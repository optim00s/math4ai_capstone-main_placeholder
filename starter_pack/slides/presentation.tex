\begin{frame}{The Central Question}

\begin{block}{Research Question}
\textit{When does a one-hidden-layer nonlinear classifier genuinely improve on a linear decision rule, and when is additional model complexity unnecessary?}
\end{block}

\vspace{0.8em}

\textbf{Our approach:}
\begin{itemize}
    \item Compare \textbf{softmax regression} (linear baseline) against a \textbf{one-hidden-layer neural network} (tanh + softmax)
    \item Three datasets of increasing complexity: Gaussian blobs, moons, digits
    \item Fair comparison: same splits, preprocessing, metrics, and checkpoint policy
    \item \textbf{More complexity is not automatically better} --- we let the evidence decide
\end{itemize}

\end{frame}

% ──────────────────────────────────────────
% SLIDE 3: Models
% ──────────────────────────────────────────
\begin{frame}{Two Models, Same Objective}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Model 1: Softmax Regression}
\[
s(x) = Wx + b
\]
\[
p_j(x) = \frac{\exp(s_j)}{\sum_\ell \exp(s_\ell)}
\]
\begin{itemize}
    \item Linear decision boundaries
    \item Parameters: $W \in \mathbb{R}^{k \times d}$, $b \in \mathbb{R}^k$
    \item Directly outputs a categorical distribution
\end{itemize}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{Model 2: Neural Network}
\[
h = \tanh(W_1 x + b_1)
\]
\[
s = W_2 h + b_2
\]
\begin{itemize}
    \item Nonlinear hidden representation
    \item \texttt{tanh} activation: smooth, zero-centered
    \item Can learn curved decision boundaries
\end{itemize}
\end{column}
\end{columns}

\vspace{0.6em}
\centering
Both trained by minimizing cross-entropy $\mathcal{L} = -\frac{1}{n}\sum_i \log p_{y_i}(x_i)$ (= negative log-likelihood)

\end{frame}

% ──────────────────────────────────────────
% SLIDE 4: Backpropagation
% ──────────────────────────────────────────
\begin{frame}{Backpropagation: Chain Rule in Action}

\textbf{Idea:} propagate loss sensitivities backward through the computation graph.

\vspace{0.3em}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\textbf{Key gradient formulas (vectorized):}
\begin{align*}
\frac{\partial \mathcal{L}}{\partial S} &= \tfrac{1}{n}(P - Y) \\[4pt]
\frac{\partial \mathcal{L}}{\partial W_2} &= \left(\tfrac{\partial \mathcal{L}}{\partial S}\right)^\top H \\[4pt]
\frac{\partial \mathcal{L}}{\partial Z_1} &= \tfrac{\partial \mathcal{L}}{\partial S}\, W_2 \odot (1 - H \odot H) \\[4pt]
\frac{\partial \mathcal{L}}{\partial W_1} &= \left(\tfrac{\partial \mathcal{L}}{\partial Z_1}\right)^\top X
\end{align*}
\end{column}
\begin{column}{0.42\textwidth}
\textbf{Verified with:}
\begin{itemize}
    \item Numerical gradient check\\($|$analytical $-$ numerical$|$ $< 10^{-4}$)
    \item Probabilities sum to 1
    \item Loss decreases on tiny batches
    \item Overfit 10 samples to 100\%
    \item No NaN/Inf in parameters
\end{itemize}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Failure Case: Under-Capacity NN ($h\!=\!2$) on 10-Class Digits}

\begin{columns}[c]
\begin{column}{0.55\textwidth}
\includegraphics[width=\textwidth]{failure_analysis.png}
\end{column}
\begin{column}{0.43\textwidth}

\begin{tabular}{lcc}
\toprule
& \textbf{Acc} & \textbf{CE} \\
\midrule
NN $h\!=\!2$ & 0.590 & 1.113 \\
NN $h\!=\!32$ & 0.957 & 0.157 \\
\bottomrule
\end{tabular}

\vspace{0.5em}
\textbf{Why it fails:}
\begin{itemize}
    \item 2 hidden units create a \textbf{2-dimensional bottleneck}
    \item 64-dim input $\to$ 2-dim $\to$ 10 classes: impossible to preserve enough structure
    \item Both train and val loss stay high $\Rightarrow$ underfitting, not overfitting
    \item \textbf{Lesson:} complexity must match task demands
\end{itemize}
\end{column}

% ──────────────────────────────────────────
% SLIDE 14: Discussion
% ──────────────────────────────────────────
\begin{frame}{Answering the Central Question}

\begin{block}{When is the linear model sufficient?}
When the class geometry is linearly separable (Gaussian blobs). Additional capacity adds no benefit.
\end{block}

\begin{block}{What representational change does the hidden layer provide?}
It transforms the input into a space where previously inseparable classes become linearly separable. On moons this means learning curved boundaries; on digits it means extracting stroke-level features.
\end{block}

\begin{block}{When does extra complexity fail to justify itself?}
When the hidden layer is too small ($h\!=\!2$ on 10 classes) --- the bottleneck destroys information. Also on linearly separable tasks, where complexity is wasted.
\end{block}

\begin{block}{What do repeated-seed statistics add?}
They reveal initialization sensitivity and give honest confidence intervals. A single run may be lucky or unlucky; five seeds show the NN advantage is consistent, not an artifact.
\end{block}

\end{frame}

% ──────────────────────────────────────────
% SLIDE 15: Limitations
% ──────────────────────────────────────────
\begin{frame}{Limitations}

\begin{itemize}
    \item \textbf{Single hidden layer only} --- deeper architectures may behave differently
    \item \textbf{Small-scale benchmark} --- 1797 samples, 64 features; conclusions may not transfer to large-scale data (e.g., ImageNet)
    \item \textbf{Only \texttt{tanh} activation} --- ReLU or other activations may yield different dynamics
    \item \textbf{Fixed regularization} ($\lambda = 10^{-4}$) --- we did not tune regularization strength
    \item \textbf{5 seeds} --- a larger seed budget would tighten confidence intervals
    \item \textbf{No test on intermediate complexity tasks} --- the gap between moons and digits leaves room for further study
    \item \textbf{PCA is linear} --- nonlinear dimensionality reduction (e.g., t-SNE) was not explored
\end{itemize}

\vspace{0.5em}
\centering\textit{These boundaries are honest --- they define what our evidence supports and what it does not.}

\end{frame}
